Taller #2: Condicionales

Enunciado 5: La Droguería Mi Salud presta sus servicios en la localidad de Suba y requiere una aplicación para poder facturar los productos que vende a sus clientes y para ello, los productos tienen unas características que deben indicársele al cliente para que pueda escoger el producto a comprar. Para cada cliente, se tienen las opciones de compra de producto, consulta de precios por producto y devoluciones en caso de que se presenten.

Código de solución: 

  Proceso punto5
    Definir opcion, aux Como Entero;
    Definir producto1, producto2, producto3, compr Como Caracter;
    Definir precioP1, precioP2, precioP3 Como Entero;
    producto1 <- "Pastillas";producto2 <- "Jarabe";producto3 <- "Vitaminas XYZ";
    precioP1 <- 15000; precioP2 <- 10000; precioP3 <- 50000;
    aux <- -1;compr <- "";
    Escribir "Men? de aplicaci?n.";
    Escribir "Seleccione una opci?n: ";
    Escribir "1) Comprar un producto.";
    Escribir "2) Consultar precio por producto.";
    Escribir "3) Devolver un producto.";
    Leer opcion;
    
    appDrogueria(opcion, aux, producto1, producto2, producto3, compr, precioP1, precioP2, precioP3);
    
  FinProceso

  SubProceso appDrogueria(opcion, aux, producto1, producto2, producto3, compr, precioP1, precioP2, precioP3)
    Segun opcion Hacer
      1:
        Escribir "Productos disponibles: ";
        Escribir "1) ", producto1, " | precio: $", precioP1;
        Escribir "2) ", producto2, " | precio: $", precioP2;
        Escribir "3) ", producto3, " | precio: $", precioP3;
        Escribir "Selecciones el producto que desea comprar: ";
        Leer aux;
        Segun aux Hacer
          1:
            Escribir "Compra realizada de ", producto1, " pagar: $", precioP1;
          2:
            Escribir "Compra realizada de ", producto2, " pagar: $", precioP2;
          3:
            Escribir "Compra realizada de ", producto3, " pagar: $", precioP3;
          De Otro Modo:
            Escribir "La opci?n no corresponde a ning?n producto!";
        FinSegun
        
      2:
        Escribir "Ingrese el nombre del producto que desea consultar: ";
        Leer compr;
        Si compr == producto1 Entonces
          Escribir "El producto: ", producto1, " tiene un precio de $", precioP1;
        SiNo
          Si compr == producto2 Entonces
            Escribir "El producto: ", producto2, " tiene un precio de $", precioP2;
          SiNo
            Si compr == producto3 Entonces
              Escribir "El producto: ", producto3, " tiene un precio de $", precioP3;
            SiNo
              Escribir "El producto: ", compr, " no se encuentra disponible en inventario.";
            FinSi
          FinSi
        FinSi			
      3:
        Escribir "Ingrese el nombre del producto que desea devolver: ";
        Leer compr;
        Si compr == producto1 Entonces
          Escribir "Reclame en caja un total de $", precioP1;
        SiNo
          Si compr == producto2 Entonces
            Escribir "Reclame en caja un total de $", precioP2;
          SiNo
            Si compr == producto3 Entonces
              Escribir "Reclame en caja un total de $", precioP3;
            SiNo
              Escribir "El producto: ", compr, " no se encuentra disponible en inventario.";
            FinSi
          FinSi
        FinSi		
      De Otro Modo:
        Escribir "La opci?n seleccionada no existe, intente de nuevo.";
    FinSegun
  FinSubProceso
